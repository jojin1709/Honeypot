    Configuration
    
      
    One file controls the whole hive.
    
      
    
        Disable a honeypot you don't need, remap a port, or turn on a feature —
        config.ini is the single source of truth, and it's the
        only file you edit for day-to-day use. Or use the Settings page in the dashboard.
      
    
      
      
        
    • Windows note — Cowrie's SSH config is synced from config.ini's ssh_port automatically on every start.
      • 
        
    • Ports under 1024 (like real port 22 or 53) need Administrator / sudo.
      • 
        
    • Local-only testing? Set bind_host = 127.0.0.1 instead of 0.0.0.0.
      • 
        
    • WSL2 note — DNS uses port 5353 (port 53 is reserved by WSL2).
      • 
      
    
    
    
    
    [general]
bind_host = 0.0.0.0
enable_ssh_cowrie = yes
enable_smb        = yes
enable_geoip      = yes
enable_log_rotation = yes
enable_notifications = no

[ports]
ssh_port    = 2222   # 22 needs admin
smb_port    = 445
dns_port    = 5353   # WSL2 uses 53
redis_port  = 6379
# ...all 17 ports configurable

[features]
abuseipdb_key =      # Threat intel API keys
virustotal_key =     # Configure in Settings page
log_max_size_mb    = 50
log_retention_days = 30
    
  
    

    



    
  
    
    ⚠
    
    
      
    Read this before you expose it to the internet.
    
      
    
        Open ports invite real traffic. Malware capture downloads real
        samples — handle them in a sandbox, never on your host. Honeypots may
        be legally regulated in some jurisdictions and will trigger security
        alerts on corporate networks. Run on an isolated VM or private network
        unless you know exactly what you're doing and have permission to do it.
